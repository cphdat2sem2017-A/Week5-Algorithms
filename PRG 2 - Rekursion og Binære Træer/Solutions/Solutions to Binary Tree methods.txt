/**
	Returns the min value in a non-empty binary search tree.
	Uses a helper method that iterates to the left to find
	the min value.
	 */
	public int minValue() {
		return(minValueRec(root, Integer.MAX_VALUE));
	}


	/**
	Finds the min value in a non-empty binary search tree.
	 */
	private int minValueRec(Node node, int min) {
		if (node == null) return min;
		else
			if (node.left == null)
				min = node.data;
			else
				min = minValueRec(node.left, min);
	
		return min;
	}
